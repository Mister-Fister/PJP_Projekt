TypeChecker

public override Type VisitFopenStatement(PJP_ProjektParser.FopenStatementContext context)
{
    // zisti typ premennej f zo SymbolTable
    var type = SymbolTables[context.ID().Symbol];

    if (type != Type.File && type != Type.Error)
    {
        Errors.ReportError(context.ID().Symbol,
            $"Variable '{context.ID().GetText()}' must be of type FILE.");
    }

    return Type.Error; // fopen nevracia zmysluplný typ
}

public override Type VisitFwriteStatement(PJP_ProjektParser.FwriteStatementContext context)
{
    var type = SymbolTables[context.ID().Symbol];

    if (type != Type.File && type != Type.Error)
    {
        Errors.ReportError(context.ID().Symbol,
            $"Variable '{context.ID().GetText()}' must be of type FILE.");
    }

    foreach (var expr in context.expression())
    {
        Visit(expr);
    }

    return Type.Error; // fwrite nevracia zmysluplný typ
}

Interpretor
switch (op)

case "open":
{
    string fileName = _stack.Pop().ToString();
    var writer = new System.IO.StreamWriter(fileName);
    _stack.Push(writer);
    break;
}

case "fwrite":
{
    int count = int.Parse(instr[1]);
    var values = new object[count];

    for (int i = count - 1; i >= 0; i--)
    {
        values[i] = _stack.Pop();
    }

    var writer = (System.IO.StreamWriter)_stack.Pop();

    foreach (var val in values)
    {
        if (val is double d)
        {
            string formatted = d.ToString(
                System.Globalization.CultureInfo.InvariantCulture);
            if (!formatted.Contains('.'))
                formatted += ".0";
            writer.Write(formatted);
        }
        else if (val is bool b)
            writer.Write(b ? "true" : "false");
        else
            writer.Write(val);
    }

    writer.WriteLine();
    writer.Flush();

    _stack.Push(writer);
    break;
}

SymbolTables.cs - nahraď existujúcu Add metódu
public void Add(IToken variable, Type type, Type elementType = Type.Error)
{
    var name = variable.Text.Trim();
    if (memory.ContainsKey(name))
    {
        Errors.ReportError(variable, $"Variable '{name}' was already declared");
    }
    else
    {
        memory.Add(name, type);
        if (type == Type.Array)
            arrayElementTypes[name] = elementType;
    }
}

SymbolTables.cs - pridaj novú metódu
public Type GetArrayElementType(IToken variable)
{
    var name = variable.Text.Trim();
    if (arrayElementTypes.ContainsKey(name))
        return arrayElementTypes[name];
    return Type.Error;
}

TypeChecker.cs - pridaj novú metódu
public override Type VisitArrayDeclaration(PJP_ProjektParser.ArrayDeclarationContext context)
{
    var typeName = context.type().GetText();
    Type type = typeName switch
    {
        "int"    => Type.Int,
        "float"  => Type.Float,
        "bool"   => Type.Bool,
        "string" => Type.String,
        _        => Type.Error
    };
    SymbolTables.Add(context.ID().Symbol, Type.Array, type);
    return Type.Error;
}

case "newarray":
{
    int size = Convert.ToInt32(_stack.Pop());
    var arr = new object[size];
    _memory[instr[1]] = arr;
    break;
}

case "astore":
{
    var value = _stack.Pop();
    int index = Convert.ToInt32(_stack.Pop());
    var arr = (object[])_stack.Pop();
    arr[index] = value;
    break;
}

case "aload":
{
    int index = Convert.ToInt32(_stack.Pop());
    var arr = (object[])_stack.Pop();
    _stack.Push(arr[index]);
    break;
